Rust syntax highlighting problems

Good: function

fn foo() {

}

Bad: function

fn foo<'a>() {

}

Good: constant

static bar: int = 5;

Bad: constant

static bar: &'static str = "hello";

Good: struct

struct Baz<T> {
    baz: T,
}

Bad: struct

struct Baz<'a> {
    baz: &'a str,
}
